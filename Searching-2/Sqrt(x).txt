class Solution {
public:
    int mySqrt(int x) {
        if(x==0)
        {
            return 0;
        }
        int l=1,h=x,ans;
       while(l<=h)
       {
           long long int mid=l+(h-l)/2;
           if(mid*mid<=x)
           {
               ans=mid;
               l=mid+1;
           }
           else
           {
               h=mid-1;
           }
           
       }
        return ans;
    }
};
